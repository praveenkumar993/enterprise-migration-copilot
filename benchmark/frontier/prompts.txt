WITH ranked_transactions AS (
  SELECT 
    customer_id,
    transaction_id,
    order_date,
    amount,
    ROW_NUMBER() OVER (PARTITION BY customer_id ORDER BY order_date DESC) AS rn,
    SUM(amount) OVER (PARTITION BY customer_id ORDER BY order_date
                      RANGE BETWEEN INTERVAL 30 DAY PRECEDING AND CURRENT ROW) AS rolling_30d
  FROM transactions
),
customer_segments AS (
  SELECT 
    customer_id,
    CASE WHEN rolling_30d > 1000 THEN 'high'
           WHEN rolling_30d > 300 THEN 'medium'
           ELSE 'low' END AS segment
  FROM ranked_transactions WHERE rn = 1
)
SELECT 
  s.segment, 
  COUNT(DISTINCT t.customer_id) AS customers, 
  SUM(t.amount) AS total
FROM 
  ranked_transactions r
JOIN 
  customer_segments s ON r.customer_id = s.customer_id
GROUP BY 
  s.segment
HAVING 
  SUM(t.amount) > 500
ORDER BY 
  total DESC;

CREATE OR REPLACE PROCEDURE get_department_stats AS
  TYPE dept_salary_rec IS RECORD (
    department_id   NUMBER,
    total_salary    NUMBER,
    avg_salary      NUMBER,
    headcount       NUMBER
  );
  TYPE dept_salary_tab IS TABLE OF dept_salary_rec INDEX BY PLS_INTEGER;
  v_dept_stats dept_salary_tab;
  i PLS_INTEGER := 1;

BEGIN
  -- Open a cursor to fetch department statistics
  OPEN c_dept_stats FOR
    SELECT department_id,
           SUM(salary) AS total_salary,
           AVG(salary) AS avg_salary,
           COUNT(*) AS headcount
    FROM employees
    GROUP BY department_id;

  -- Fetch data into the table variable using BULK COLLECT
  LOOP
    FETCH c_dept_stats BULK COLLECT INTO v_dept_stats;
    EXIT WHEN v_dept_stats.COUNT = 0;
    FOR j IN 1 .. v_dept_stats.COUNT LOOP
      v_dept_stats(j).department_id := rec.department_id;
      v_dept_stats(j).total_salary  := rec.total_salary;
      v_dept_stats(j).avg_salary    := rec.avg_salary;
      v_dept_stats(j).headcount     := rec.headcount;
    END LOOP;
    i := i + v_dept_stats.COUNT;
  END LOOP;

  -- Close the cursor
  CLOSE c_dept_stats;

  -- Handle any exceptions that occur during the procedure execution
  BEGIN

  WHERE sensitivity = p_sensitivity AND active_flag = 'Y'

  BULK COLLECT INTO v_suspicious
  FROM payment_transactions
  WHERE txn_date = p_detection_date AND flagged = 'N'

CREATE PROCEDURE usp_TemporalDataReconstructWithMergeAndCursors
    @EntityId INT,
    @AsOfDate DATETIME,
    @EntityType NVARCHAR(50)
AS
BEGIN
    SET NOCOUNT ON;

    -- Create a temporary table to store the most recent values for each field
    CREATE TABLE #TemporalData (
        EntityId INT,
        FieldName NVARCHAR(50),
        OldValue NVARCHAR(MAX),
        NewValue NVARCHAR(MAX),
        ChangedAt DATETIME
    );

    -- MERGE statement to update existing records or insert new ones
    BEGIN TRY
        MERGE INTO #TemporalData t
        USING (
            SELECT 
                EntityId,
                FieldName,
                OldValue,
                NewValue,
                ChangedAt
            FROM AuditChanges
            WHERE EntityId = @EntityId
            AND EntityType = @EntityType
            AND ChangedAt <= @AsOfDate
        ) s ON t.EntityId = s.EntityId AND t.FieldName = s.FieldName

        WHEN MATCHED THEN
            UPDATE SET 
                t.OldValue = s.OldValue,
                t.NewValue = s.NewValue,
                t.ChangedAt = s.ChangedAt
        WHEN NOT MATCHED BY TARGET THEN
            INSERT (EntityId, FieldName, OldValue, NewValue, ChangedAt)
            VALUES (s.EntityId, s.FieldName, s.OldValue, s.NewValue, s.ChangedAt);
    END TRY
    BEGIN CATCH
        -- Handle errors that occur during the MERGE statement
        DECLARE @ErrorMessage NVARCHAR(MAX);
        SELECT @ErrorMessage = ERROR_MESSAGE();
        RAISERROR ('An error occurred: %s', 16, 1, @ErrorMessage);
    END CATCH

    -- Fetch the most recent values for each field
    WITH PointInTime AS (
        SELECT 
            EntityId,
            FieldName,
            NewValue AS CurrentValue,
            ChangedAt
        FROM #TemporalData
    )

    SELECT 
        e.EntityId,
        @EntityType AS EntityType,
        @AsOfDate AS AsOfDate,
        p.FieldName,
        p.CurrentValue,
        p.ChangedAt AS LastModified,
        COUNT(al.ChangeId) OVER (PARTITION BY al.EntityId) AS TotalChangeCount
    FROM PointInTime p
    CROSS JOIN (SELECT DISTINCT EntityId FROM AuditChanges WHERE EntityId = @EntityId) e
    JOIN AuditChanges al ON al.EntityId = @EntityId
    ORDER BY p.FieldName;
END
